### ITEM 1 - GENERACION DE TABLAS
- Creación de tablas 
```SQL
CREATE TABLE COMUNA(
    ID_COMUNA      NUMBER(3) PRIMARY KEY,
    NOM_COMUNA     VARCHAR2(40) NOT NULL
);

CREATE TABLE MARCA(
    ID_MARCA       NUMBER(5) PRIMARY KEY,
    NOM_MARCA      VARCHAR2(40) NOT NULL
);

CREATE TABLE CATEGORIA(
    ID_CATEGORIA    NUMBER(3) PRIMARY KEY,
    NOM_CATEGORIA   VARCHAR2(40) NOT NULL,
    PORCENTAJE      NUMBER(5,2) NOT NULL
);

CREATE TABLE ZONA(
    ID_ZONA      NUMBER(3) PRIMARY KEY,
    NOM_ZONA     VARCHAR2(30) NOT NULL
);

CREATE TABLE CLIENTE(
    ID_CLIENTE      NUMBER(5) PRIMARY KEY,
    APE_PAT         VARCHAR2(30) NOT NULL,
    APE_MAT         VARCHAR2(30),
    NOMBRE          VARCHAR2(40) NOT NULL,
    DIRECCION       VARCHAR2(80) NOT NULL,
    TELEFONO        VARCHAR2(15),
    ID_COMUNA       NUMBER(3),
    CONSTRAINT fk_cliente_comuna
        FOREIGN KEY (ID_COMUNA)
        REFERENCES COMUNA(ID_COMUNA)
);

CREATE TABLE ARTICULO(
    ID_ARTICULO        NUMBER(5) PRIMARY KEY,
    NOM_ARTICULO       VARCHAR2(40) NOT NULL,
    PRECIO_UNITARIO    NUMBER(10) NOT NULL,
    STOCK              NUMBER(10) NOT NULL,
    ID_CATEGORIA       NUMBER(3),
    CONSTRAINT fk_articulo_categoria
        FOREIGN KEY (ID_CATEGORIA)
        REFERENCES CATEGORIA(ID_CATEGORIA)
);

CREATE TABLE VENDEDOR(
    ID_VENDEDOR      NUMBER(3) PRIMARY KEY,
    RUT_VENDEDOR     VARCHAR2(12) NOT NULL,
    NOMBRE           VARCHAR2(40) NOT NULL,
    PATERNO          VARCHAR2(30) NOT NULL,
    MATERNO          VARCHAR2(30),
    FECHA_NAC        DATE NOT NULL,
    FEC_CONTRATO     DATE NOT NULL,
    SUELDO           NUMBER(10) NOT NULL,
    COMISION         NUMBER(4,2) NOT NULL,
    ID_ZONA          NUMBER(3),
    ID_CATEGORIA     NUMBER(3),
    CONSTRAINT fk_vendedor_zona
        FOREIGN KEY (ID_ZONA)
        REFERENCES ZONA(ID_ZONA),
    CONSTRAINT fk_vendedor_categoria
        FOREIGN KEY (ID_CATEGORIA)
        REFERENCES CATEGORIA(ID_CATEGORIA)
);

CREATE TABLE FACTURA(
    ID_FACTURA      NUMBER(10) PRIMARY KEY,
    FECHA           DATE NOT NULL,
    ID_CLIENTE      NUMBER(5),
    ID_VENDEDOR     NUMBER(3),
    CONSTRAINT fk_factura_cliente
        FOREIGN KEY (ID_CLIENTE)
        REFERENCES CLIENTE(ID_CLIENTE),
    CONSTRAINT fk_factura_vendedor
        FOREIGN KEY (ID_VENDEDOR)
        REFERENCES VENDEDOR(ID_VENDEDOR)
);

CREATE TABLE DETALLEFACTURA(
    ID_DETALLE       NUMBER(10) PRIMARY KEY,
    ID_FACTURA       NUMBER(10),
    ID_ARTICULO      NUMBER(5),
    CANTIDAD         NUMBER(10) NOT NULL,
    CONSTRAINT fk_detalle_factura
        FOREIGN KEY (ID_FACTURA)
        REFERENCES FACTURA(ID_FACTURA),
    CONSTRAINT fk_detalle_articulo
        FOREIGN KEY (ID_ARTICULO)
        REFERENCES ARTICULO(ID_ARTICULO)
);

```

### ITEM II - CONSULTAS SQL
Desarrollar una consulta que muestre los datos de vendedores, incluyendo:
- Sueldo incrementado en un **20%**
- Una columna con un **bono adicional de $500.000**
- Una tercera columna llamada **"Sueldo Final"**
- Orden
- ado de **mayor a menor** por sueldo final
```sql
SELECT 
    ID_VENDEDOR,
    NOMBRE,
    PATERNO,
    MATERNO,
    SUELDO,
    SUELDO * 1.20 AS SUELDO_CON_AUMENTO,
    500000 AS BONO,
    (SUELDO * 1.20) + 500000 AS "Sueldo Final"
FROM VENDEDOR
ORDER BY "Sueldo Final" DESC;
```
- Concatenar:
	- Nombre
	- Paterno
	- Materno

```SQL
SELECT 
    ID_VENDEDOR,
    (NOMBRE || ' ' || PATERNO || ' ' || NVL(MATERNO, '')) AS "Nombre Completo",
    SUELDO,
    SUELDO * 1.20 AS SUELDO_CON_AUMENTO,
    500000 AS BONO,
    (SUELDO * 1.20) + 500000 AS "Sueldo Final"
FROM VENDEDOR
ORDER BY "Sueldo Final" DESC;
```
